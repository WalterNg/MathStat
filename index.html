    st.line_chart(data)

# Dynamic tabs from data
sections = ['Overview', 'Details', 'Reports']
tabs = st.tabs(sections)

for i, tab in enumerate(tabs):
    with tab:
        st.write(f"Content for {sections[i]}")
        if i == 0:
            st.bar_chart(data)
        elif i == 1:
            st.table(details_df)

Custom Button Navigation

import streamlit as st

# Custom navigation with session state
if 'active_tab' not in st.session_state:
    st.session_state.active_tab = 'Home'

cols = st.columns(3)
buttons = ['Home', 'About', 'Contact']

for i, button in enumerate(buttons):
    with cols[i]:
        if st.button(button, key=f"nav_{button}"):
            st.session_state.active_tab = button

# Display content based on active tab
if st.session_state.active_tab == 'Home':
    st.write("Home content")
elif st.session_state.active_tab == 'About':
    st.write("About content")
else:
    st.write("Contact content")

Best Practices

Guidelines and recommendations for tab implementation

User Experience Guidelines

  • Keep tab labels concise and descriptive
  • Use icons to improve visual hierarchy
  • Ensure mobile responsiveness
  • Maintain consistent styling across tabs
  • Test accessibility with screen readers
  • Consider loading states for heavy content

Performance Considerations

  • Lazy load content in inactive tabs
  • Use session state for tab persistence
  • Minimize CSS injection frequency
  • Cache heavy computations
  • Consider component rerun behavior

Accessibility Best Practices

  • Ensure proper ARIA labels
  • Maintain keyboard navigation
  • Provide sufficient color contrast
  • Use semantic HTML structure
  • Test with screen readers
